# UI
User Interface Modules for Roblox


## Colors
This module is generated by the following code
```lua
local String = "local rgb = Color3.fromRGB\n\nlocal Black = rgb(0, 0, 0);\nlocal White = rgb(255, 255, 255);"
local Return = "\n\nreturn {"
local Page = game:GetService("HttpService"):GetAsync("https://material.io/guidelines/style/color.html")

for Group, Name in Page:gmatch("(<section class=\"color%-group\">.-<span class=\"name.-\">([%w ]+)</span>.-</section>)") do
    Name = Name:gsub("%s", "")
    local List = "\n\nlocal " .. Name .. " = {"
    local Accent
    for Hex1, Hex2, Hex3, Color in Group:gmatch("<li class=\"color.-style=\"background%-color: #(%w%w)(%w%w)(%w%w);\"><span class=\"shade.-\">(%w+)</span><span class=\"hex\">#%w+</span></li>") do
        if type(tonumber(Color)) == "number" then
            List = ("%s\n\t[%s] = rgb(%u, %u, %u);"):format(List, Color, "0x" .. Hex1, "0x" .. Hex2, "0x" .. Hex3)
        else
            if not Accent then
                Accent = "\n\tAccent = {"
            end
            Accent = ("%s\n\t\t[%s] = rgb(%u, %u, %u);"):format(Accent, Color:sub(2), "0x" .. Hex1, "0x" .. Hex2, "0x" .. Hex3)
        end
    end
    String = String .. List .. (Accent and "\n" .. Accent .. "\n\t};" or "") .. "\n}"
    Return = ("%s\n\t%s = %s;"):format(Return, Name, Name)
end

print(String .. Return .. "\n\tBlack = Black;\n\tWhite = White;\n}\n")
```
